582 template<typename ReturnT, typename ...ChannelTs>
583 Event::SharedPtr Modflow::prepareEmit (const Channel &channel, const Module *caller)
584 {
585  if (!channel.checkType<ChannelTs...> ())
586  throw std::runtime_error (errorChannelTypeMismatch<ChannelTs...> (channel, caller, true));
587 
588  if (!channel.checkOwnership(caller))
589  throw std::runtime_error (errorOwnership (channel, caller));
590 
591  Event::SharedPtr lastEvent = caller->lastEvent();
592  Event::SharedPtr event;
593 
594  if (lastEvent == nullptr)
595  event = std::make_shared<Event> (caller, &channel);
596  else
597  event = std::make_shared<Event> (lastEvent, caller, &channel);
598 
599  if (debugFilters (event))
600  debugTrackEmit (event->depth(), channel, caller, _connections[channel.id()].size ());
601 
602  return event;
603 }

624 inline bool Modflow::debugFilters (const Event::SharedPtr &event)
625 {
626  if (!_debug.enabled)
627  return false;
628 
629  for (const std::string &curr : _debug.filterOnlyChannels) {
630  if (!event->channelInAncestors (curr))
631  return false;
632  }
633 
634  for (const std::string &curr : _debug.filterOnlyModules) {
635  if (!event->moduleInAncestors (curr))
636  return false;
637  }
638 
639  for (const std::string &curr : _debug.filterExcludeChannels) {
640  if (event->channelInAncestors (curr))
641  return false;
642  }
643 
644  for (const std::string &curr : _debug.filterExcludeModules) {
645  if (event->moduleInAncestors (curr))
646  return false;
647  }
648 
649  return true;
650 }
