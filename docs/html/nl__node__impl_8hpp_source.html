22 template<typename ParameterT>
23 std::any ROS2ParameterServer::processResolved (Call callType, const std::string &name, const std::optional<std::any> &defaultValue)
24 {
25  switch (callType) {
26  case Call::DECLARE:
27  if (defaultValue.has_value())
28  return _node->declare_parameter<ParameterT> (name, std::any_cast<ParameterT> (*defaultValue));
29  else
30  return _node->declare_parameter<ParameterT> (name);
31  break;
32  case Call::GET:
33  auto param = _node->get_parameter (name).get_value<ParameterT> ();
34  // ROS2 stores parameters in 64bits, so if you ask e.g. a float instead of a double, it will return a double. So if you ask for a float, it needs to be converted
35  if constexpr (internal::traits::is_container<ParameterT>::value) // convert vector to correct type (e.g. float instead of double)
36  return ParameterT (param.begin (), param.end ());
37  else
38  return static_cast<ParameterT> (param);
39  }
40 
41  return std::any ();
42 }

53 std::any ROS2ParameterServer::resolveOne(const std::type_index &type, Call callType, const std::string &name, const std::optional<std::any> &defaultValue) {

107 bool RosConfiguration::waitAlive ()
108 {
109  using namespace std::chrono_literals;
110 
111  constexpr auto TIMEOUT = 5s;
112 
113  auto client = _node->create_client<std_srvs::srv::Empty>("/ros_configuration_alive");
114  auto request = std::make_shared<std_srvs::srv::Empty::Request> ();

116  if (!client->wait_for_service(TIMEOUT))
